Scene Elements

Scene Element 1: Basic Shapes, Lighting, Textures, Shadows, and Motion


// Create basic shapes
var box = BABYLON.MeshBuilder.CreateBox("box", { size: 2 }, scene);
var sphere = BABYLON.MeshBuilder.CreateSphere("sphere", { diameter: 2 }, scene);

// Apply textures and materials
var boxMaterial = new BABYLON.StandardMaterial("boxMaterial", scene);
boxMaterial.diffuseTexture = new BABYLON.Texture("textures/boxTexture.jpg", scene);
box.material = boxMaterial;

var sphereMaterial = new BABYLON.StandardMaterial("sphereMaterial", scene);
sphereMaterial.diffuseTexture = new BABYLON.Texture("textures/sphereTexture.jpg", scene);
sphere.material = sphereMaterial;

// Shadows
var shadowGenerator = new BABYLON.ShadowGenerator(1024, light);
shadowGenerator.addShadowCaster(box);
shadowGenerator.addShadowCaster(sphere);

// Animation
var animationBox = new BABYLON.Animation("boxAnimation", "rotation.y", 30, BABYLON.Animation.ANIMATIONTYPE_FLOAT, BABYLON.Animation.ANIMATIONLOOPMODE_CYCLE);
var keysBox = [];
keysBox.push({ frame: 0, value: 0 });
keysBox.push({ frame: 100, value: Math.PI * 2 });
animationBox.setKeys(keysBox);
box.animations = [animationBox];
